```java
// ========== 第 37-81 行：核心过滤方法 ==========

// 第 37-40 行：重写doFilterInternal方法，这是过滤器的核心方法
// HttpServletRequest request：请求对象，包含客户端发送的所有信息
// HttpServletResponse response：响应对象，用于向客户端返回数据
// FilterChain filterChain：过滤器链，用于将请求传递给下一个过滤器
@Override
protected void doFilterInternal(HttpServletRequest request, HttpServletResponse response, FilterChain filterChain) {
    
    // 第 42-44 行：异常处理，确保即使出现异常也不会中断请求处理
    try {
        
        // 第 47-49 行：获取JWT Token
        // getJwtFromRequest(request) 从HTTP请求头中提取Token
        // 将Token存储在局部变量中，方便后续使用
        String jwt = getJwtFromRequest(request);

        // 第 51-72 行：验证JWT Token并设置用户认证
        // StringUtils.hasText(jwt)：检查jwt是否不为空且包含非空白字符
        // jwtUtil.validateToken(jwt)：验证Token的签名和有效期
        if (StringUtils.hasText(jwt) && jwtUtil.validateToken(jwt)) {
            
            // 第 54 行：解析Token获取用户信息
            // parseToken方法返回一个Claims对象，包含了Token中存储的所有数据
            Claims claims = jwtUtil.parseToken(jwt);
            
            // 第 55-57 行：从Claims中提取用户信息
            // get("userId")：获取存储在Token中的用户ID
            // claims.get("username")：获取用户名
            // claims.get("role")：获取用户角色
            Integer userId = (Integer) claims.get("userId");
            String username = (String) claims.get("username");
            String role = (String) claims.get("role");

            // 第 59-65 行：检查用户状态，确保用户没有被禁用
            // usersService.getById(userId)：根据ID查询用户信息
            Users user = usersService.getById(userId);
            
            // 第 60-64 行：判断用户是否存在且状态为active
            // user == null：用户不存在（如被删除）
            // !"active".equals(user.getStatus())：用户状态不是active（被禁用/锁定）
            if (user == null || !"active".equals(user.getStatus())) {
                // 第 61 行：清空SecurityContext，防止未授权访问
                SecurityContextHolder.clearContext();
                // 第 62 行：返回403 Forbidden状态码
                response.setStatus(HttpServletResponse.SC_FORBIDDEN);
                // 第 63 行：直接返回，不继续处理请求
                return;
            }

            // 第 67-69 行：创建权限列表
            // ArrayList：创建可变数组用于存储权限
            List<SimpleGrantedAuthority> authorities = new ArrayList<>();
            
            // 第 68 行：为用户添加角色权限
            // "ROLE_" + role.toUpperCase()：Spring Security要求角色前面加"ROLE_"前缀
            // toUpperCase()：将角色名转换为大写，保持一致性
            authorities.add(new SimpleGrantedAuthority("ROLE_" + role.toUpperCase()));

            // 第 70-72 行：创建认证令牌并设置到SecurityContext
            // new UsernamePasswordAuthenticationToken(...)：创建认证令牌
            // 第一个参数：用户主键（通常为用户名或用户对象）
            // 第二个参数：用户凭证（密码等，设为null表示不保留）
            // 第三个参数：用户的权限列表
            UsernamePasswordAuthenticationToken authentication = 
                new UsernamePasswordAuthenticationToken(username, null, authorities);
            
            // 第 72 行：将认证信息设置到SecurityContext中
            // 这样后续代码可以通过SecurityContextHolder.getContext()获取当前用户信息
            SecurityContextHolder.getContext().setAuthentication(authentication);
        }
        
    } catch (Exception ex) {
        // 第 76-78 行：捕获并记录异常
        // 即使JWT验证失败，也要继续处理请求，避免服务中断
        logger.error("无法设置用户认证：" + ex.getMessage(), ex);
    }

    // 第 80-81 行：将请求传递给过滤器链中的下一个过滤器
    // 这是非常重要的一步，否则请求会被阻塞
    filterChain.doFilter(request, response);
}
```

```java
// 第 88-94 行：从HTTP请求头中提取JWT Token的私有方法
private String getJwtFromRequest(HttpServletRequest request) {
    
    // 第 90 行：获取HTTP请求头中的Authorization字段
    // Authorization字段通常包含 "Bearer <token>" 格式
    String bearerToken = request.getHeader("Authorization");
    
    // 第 91 行：检查Token是否有效
    // StringUtils.hasText：检查是否非空且包含非空白字符
    // startsWith("Bearer ")：检查是否以"Bearer "开头（注意后面有空格）
    if (StringUtils.hasText(bearerToken) && bearerToken.startsWith("Bearer ")) {
        
        // 第 93 行：截取Token部分（去掉"Bearer "前缀）
        // substring(7)：从第7个字符开始截取，即去掉前6个字符"Bearer "
        return bearerToken.substring(7);
    }
    
    // 第 95 行：如果不满足条件，返回null
    return null;
}
```

**功能总结**：
1. 拦截每个HTTP请求
2. 从请求头获取JWT Token
3. 验证Token有效性
4. 检查用户状态（是否被禁用）
5. 设置Spring Security上下文
6. 传递给下一个过滤器

```java
// ========== 第 37-94 行：环绕通知 - 核心权限检查逻辑 ==========

// 第 37-40 行：定义环绕通知
// @Around：环绕通知，在方法执行前后都执行
// "@annotation(RequiresRoles)"：匹配带有@RequiresRoles注解的方法
@Around("@annotation(com.example.demo.security.RequiresRoles) || @within(com.example.demo.security.RequiresRoles)")
public Object checkRole(ProceedingJoinPoint joinPoint) throws Throwable {
    
    // 第 43-45 行：获取当前HTTP请求
    // RequestContextHolder：Spring提供的请求上下文持有者
    // getRequestAttributes()：获取请求属性
    // cast为ServletRequestAttributes：转换为Servlet请求属性
    ServletRequestAttributes attributes = (ServletRequestAttributes) RequestContextHolder.getRequestAttributes();
    
    // 第 46 行：获取HttpServletRequest对象
    HttpServletRequest request = attributes.getRequest();
    
    // 第 49-55 行：获取当前用户的角色
    // getCurrentUserRole(request)：从请求中解析用户角色（私有方法）
    String role = getCurrentUserRole(request);
    
    // 第 57-61 行：检查用户是否已登录
    // role == null：表示用户未登录或Token无效
    if (role == null) {
        // 抛出未授权异常，中断请求
        throw new UnauthorizedException("未登录或登录已过期，请重新登录");
    }
    
    // 第 64-66 行：获取方法签名信息
    // joinPoint.getSignature()：获取方法签名
    // cast为MethodSignature：用于获取方法详细信息
    MethodSignature signature = (MethodSignature) joinPoint.getSignature();
    
    // 第 67 行：获取被拦截的方法对象
    Method method = signature.getMethod();
    
    // 第 69-71 行：获取方法上的@RequiresRoles注解
    // getAnnotation：获取指定类型的注解
    // 首先检查方法级别，再检查类级别
    RequiresRoles methodRequiresRoles = method.getAnnotation(RequiresRoles.class);
```

```java
// ========== 第 73-85 行：权限判断逻辑 ==========

    // 第 73 行：从注解中获取需要的角色数组
    // 例如：@RequiresRoles({"admin", "super_admin"}) 
    // 则requiredRoles = ["admin", "super_admin"]
    String[] requiredRoles = requiresRoles.value();
    
    // 第 74 行：定义权限检查结果变量
    boolean hasPermission = false;
    
    // 第 76-80 行：根据注解的logical属性判断权限
    if (requiresRoles.logical() == RequiresRoles.Logical.OR) {
        // OR逻辑：用户拥有任一所需角色即可访问
        // Stream流式处理，遍历所有所需角色
        // anyMatch：只要有一个匹配就返回true
        hasPermission = Arrays.stream(requiredRoles)
                .anyMatch(requiredRole -> requiredRole.equals(role));
        // 例如：用户角色为"admin"，requiredRoles为["admin", "super_admin"]
        // anyMatch会检查"admin"是否等于"admin"，结果为true，用户有权限
    } else {
        // AND逻辑：用户必须拥有所有所需角色
        // allMatch：所有都匹配才返回true
        hasPermission = Arrays.stream(requiredRoles)
                .allMatch(requiredRole -> requiredRole.equals(role));
        // 例如：用户角色为"admin"，requiredRoles为["admin", "super_admin"]
        // "admin"不等于"super_admin"，allMatch结果为false，用户无权限
    }
    
    // 第 82-85 行：如果没有权限，抛出异常
    if (!hasPermission) {
        // 抛出未授权异常，中断请求
        // 这个异常会被GlobalExceptionHandler捕获并返回统一格式的错误响应
        throw new UnauthorizedException();
    }
    
    // 第 87-89 行：权限检查通过，继续执行目标方法
    // joinPoint.proceed()：执行被拦截的方法
    // 返回方法的执行结果
    return joinPoint.proceed();
}
```

```java
// ========== 第 101-120 行：获取当前用户角色 ==========

// 第 101-120 行：私有方法，从请求中获取当前用户角色
private String getCurrentUserRole(HttpServletRequest request) {
    
    // 第 103-109 行：从SecurityContext获取已认证的用户信息
    // getContext()：获取安全上下文
    // getAuthentication()：获取认证信息（在上面JwtAuthenticationFilter中设置的）
    Authentication authentication = SecurityContextHolder.getContext().getAuthentication();
    
    // 第 110-113 行：检查认证信息是否有效
    if (authentication != null && authentication.isAuthenticated()) {
        // 第 112 行：获取用户的权限列表
        // getAuthorities()：获取用户的权限集合
        Collection<? extends GrantedAuthority> authorities = authentication.getAuthorities();
        
        // 第 113-117 行：遍历权限列表，找到第一个角色
        // 这里假设用户只有一个角色
        for (GrantedAuthority authority : authorities) {
            // 获取权限字符串，如"ROLE_admin"
            String authorityStr = authority.getAuthority();
            // 去掉"ROLE_"前缀，得到实际角色名
            if (authorityStr.startsWith("ROLE_")) {
                return authorityStr.substring(5);  // 去掉"ROLE_"后返回
            }
        }
    }
    
    // 第 119-120 行：如果无法获取角色，尝试从JWT Token中获取
    // 这是一个备选方案，处理SecurityContext中没有数据的情况
    String bearerToken = request.getHeader("Authorization");
    if (StringUtils.hasText(bearerToken) && bearerToken.startsWith("Bearer ")) {
        String token = bearerToken.substring(7);  // 去掉"Bearer "前缀
        if (jwtUtil.validateToken(token)) {  // 验证Token
            Claims claims = jwtUtil.parseToken(token);  // 解析Token
            return (String) claims.get("role");  // 获取角色
        }
    }
    
    // 第 122 行：如果都获取不到，返回null
    return null;
}
```

**功能总结**：
1. 拦截所有带 `@RequiresRoles` 注解的方法
2. 从 SecurityContext 或 JWT Token 获取当前用户角色
3. 与注解要求的角色进行匹配
4. 支持 OR/AND 两种匹配逻辑
5. 权限不足时抛出异常

### 1.3 用户登录

**代码位置**：`zina-tch/src/main/java/com/example/demo/controller/AuthController.java`

#### 完整代码逐行解析

```java
// ========== 第 48-116 行：登录方法 ==========

// 第 48-50 行：定义POST请求的登录接口
// @PostMapping("/login")：处理POST请求，路径为/api/auth/login
// @RequestBody Map<String, String>：将JSON请求体映射为Map
@PostMapping("/login")
public ResponseEntity<Result<Map<String, Object>>> login(@RequestBody Map<String, String> loginDto) {
    
    // 第 51-52 行：从请求体中获取用户名和密码
    // get("username")：根据键获取值
    String username = loginDto.get("username");
    String password = loginDto.get("password");
    
    // 第 54-57 行：参数校验 - 检查用户名和密码是否为空
    // username == null || password == null：任一参数为空
    if (username == null || password == null) {
        // ResponseEntity.badRequest()：返回400状态码
        // Result.validateFailed()：创建验证失败的统一响应格式
        return ResponseEntity.badRequest()
                .body(Result.validateFailed("用户名或密码不能为空"));
    }
    
    // 第 59-67 行：构建查询条件，查询用户
    // 支持三种登录方式：邮箱、手机号、用户名
    QueryWrapper<Users> queryWrapper = new QueryWrapper<>();
    
    // 第 60-66 行：使用and()构建复杂查询条件
    // wrapper.and()：在括号内使用and连接多个条件
    queryWrapper.and(wrapper -> wrapper
            // eq("email", username)：email字段等于username
            .eq("email", username)
            // .or()：或者
            .or()
            // eq("phone", username)：phone字段等于username
            .eq("phone", username)
            .or()
            // eq("name", username)：name字段等于username
            .eq("name", username));
    // 最终SQL：(email = ? OR phone = ? OR name = ?)
    
    // 第 68 行：执行查询，获取用户对象
    // usersService.getOne()：根据条件查询一条记录
    Users user = usersService.getOne(queryWrapper);
    
    // 第 70-72 行：检查用户是否存在
    // user == null：没有找到匹配的用户
    if (user == null) {
        // 返回400状态码，提示用户名或密码错误（不明确指出是哪个错误）
        return ResponseEntity.badRequest()
                .body(Result.validateFailed("用户名或密码错误"));
    }

    // 第 74-77 行：校验密码
    // passwordUtil.matches()：使用BCrypt验证密码
    // 第一个参数是用户输入的明文密码
    // 第二个参数是数据库中存储的加密密码
    // BCrypt会自动从第二个参数中提取盐值，对第一个参数进行加密后比较
    if (!passwordUtil.matches(password, user.getPassword())) {
        return ResponseEntity.badRequest()
                .body(Result.validateFailed("用户名或密码错误"));
    }

    // 第 79-94 行：检查用户状态
    // !"active".equals(user.getStatus())：用户状态不是active
    if (!"active".equals(user.getStatus())) {
        // 根据不同状态返回不同的错误信息
        if ("inactive".equals(user.getStatus())) {
            return ResponseEntity.status(HttpStatus.FORBIDDEN)  // 403状态码
                    .body(Result.error(403, "该用户被禁用"));
        } else if ("locked".equals(user.getStatus())) {
            return ResponseEntity.status(HttpStatus.FORBIDDEN)
                    .body(Result.error(403, "该用户已被锁定"));
        } else if ("banned".equals(user.getStatus())) {
            return ResponseEntity.status(HttpStatus.FORBIDDEN)
                    .body(Result.error(403, "该用户已被封禁"));
        } else {
            // 其他未知状态
            return ResponseEntity.status(HttpStatus.FORBIDDEN)
                    .body(Result.error(403, "该用户状态异常，请联系管理员"));
        }
    }

    // 第 96-98 行：更新最后登录时间
    user.setLastLogin(new Date());  // new Date()：获取当前时间
    usersService.updateById(user);  // 根据ID更新用户信息

    // 第 100-101 行：生成JWT Token
    // jwtUtil.generateToken()：生成包含用户信息的Token
    // 参数：用户ID、用户名、用户角色
    String token = jwtUtil.generateToken(user.getId(), user.getName(), user.getRole());
    
    // 第 103-105 行：创建返回结果Map
    Map<String, Object> result = new HashMap<>();
    result.put("token", token);  // JWT Token，前端需要存储
    result.put("userId", user.getId());  // 用户ID
    result.put("name", user.getName());  // 用户名
    result.put("role", user.getRole());  // 用户角色

    // 第 107-114 行：返回会员信息（如果用户是会员）
    result.put("memberNumber", user.getMemberNumber());  // 会员编号
    result.put("membershipType", user.getMembershipType());  // 会员类型
    result.put("membershipStartDate", user.getMembershipStartDate());  // 会员开始时间
    result.put("membershipExpireDate", user.getMembershipExpireDate());  // 会员到期时间
    result.put("isValidMember", isValidMember(user));  // 是否有效会员

    // 第 116 行：返回成功响应
    // ResponseEntity.ok()：返回200状态码
    // Result.success(result, "登录成功")：创建成功的统一响应
    return ResponseEntity.ok(Result.success(result, "登录成功"));
}
```

### 1.4 会员状态判断

**代码位置**：`zina-tch/src/main/java/com/example/demo/controller/AuthController.java`

#### 完整代码逐行解析

```java
// ========== 第 176-185 行：会员状态判断方法 ==========

// 第 176-177 行：私有方法，用于判断用户是否为有效会员
// private：此类私有方法，只能在AuthController内部调用
private boolean isValidMember(Users user) {
    
    // 第 178 行：空值检查
    // 防止user对象为null时调用其方法导致NullPointerException
    if (user == null) return false;
    
    // 第 179 行：检查是否为管理员
    // "admin".equals(user.getRole())：检查角色是否为admin
    // "super_admin".equals(user.getRole())：检查角色是否为super_admin
    // 管理员始终有权限，不受会员限制
    if ("admin".equals(user.getRole()) || "super_admin".equals(user.getRole())) {
        return true;  // 管理员直接返回true
    }
    
    // 第 180-184 行：检查是否为会员
    // "member".equals(user.getRole())：检查角色是否为member
    if ("member".equals(user.getRole())) {
        // 第 181-183 行：检查会员到期时间
        // user.getMembershipExpireDate() != null：到期时间不为空
        // user.getMembershipExpireDate().after(new Date())：到期时间在当前时间之后
        // after()方法：检查此日期是否在指定日期之后
        return user.getMembershipExpireDate() != null
                && user.getMembershipExpireDate().after(new Date());
    }
    
    // 第 185 行：默认返回false
    // 普通用户（role为"user"）或其他未知角色都返回false
    return false;
}
```

**功能总结**：
1. 管理员（admin/super_admin）始终返回 true
2. 会员（member）需检查到期时间是否有效
3. 普通用户和其他角色返回 false

```java
// 第 45-47 行：私有方法，从HTTP请求中提取用户ID
private Long extractUserIdFromAuthHeader(HttpServletRequest request) {
    
    // 第 48 行：获取Authorization请求头
    String bearerToken = request.getHeader("Authorization");
    
    // 第 49-52 行：检查Token格式
    // StringUtils.hasText：检查是否非空
    // startsWith("Bearer ")：检查是否以"Bearer "开头
    if (StringUtils.hasText(bearerToken) && bearerToken.startsWith("Bearer ")) {
        
        // 第 54 行：提取Token字符串
        String token = bearerToken.substring(7);
        
        // 第 55-57 行：验证并解析Token
        try {
            if (jwtUtil.validateToken(token)) {  // 验证Token有效性
                Integer userId = jwtUtil.getUserIdFromToken(token);  // 从Token中获取用户ID
                if (userId != null) {
                    log.info("从JWT Token提取用户ID: {}", userId);
                    return userId.longValue();  // 转换为Long类型返回
                }
            }
        } catch (Exception e) {
            log.warn("从JWT Token解析用户ID失败: {}", e.getMessage());
        }
    }
    
    // 第 62 行：获取失败返回null
```

```java
// ========== 第 50-67 行：判断用户是否为有效会员 ==========

// 第 50-52 行：重写isValidMember方法
// @Override：表示这是对接口或父类方法的重写
@Override
public boolean isValidMember(Users user) {
    
    // 第 54-55 行：空值检查
    // 防止user对象为null时调用其方法导致NullPointerException
    if (user == null) {
        return false;  // 空用户直接返回false
    }
    
    // 第 57-60 行：检查是否为管理员
    // 管理员角色不受会员限制，可以访问所有资源
    // "admin".equals(user.getRole())：检查是否管理员
    // "super_admin".equals(user.getRole())：检查是否超级管理员
    if ("admin".equals(user.getRole()) || "super_admin".equals(user.getRole())) {
        return true;  // 管理员直接返回true，有权限
    }
    
    // 第 62-67 行：检查是否为有效会员
    // 只有角色为"member"的用户才需要检查到期时间
    if ("member".equals(user.getRole())) {
        // 第 63-65 行：检查会员到期时间是否有效
        
        // getMembershipExpireDate()：获取会员到期时间
        // 如果为null，表示没有设置到期时间（数据异常）
        if (user.getMembershipExpireDate() == null) {
            return false;  // 到期时间为null，返回false
        }
        
        // getMembershipExpireDate().after(new Date())：
        // after()方法检查到期时间是否在当前时间之后
        // 如果返回true，表示会员尚未过期
        // 如果返回false，表示会员已过期
        return user.getMembershipExpireDate().after(new java.util.Date());
    }
    
    // 第 69 行：其他角色返回false
    // 普通用户（role为"user"）或其他未知角色都返回false
    return false;
}
```
